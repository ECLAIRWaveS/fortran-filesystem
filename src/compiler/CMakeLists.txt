if(CMAKE_Fortran_COMPILER_ID STREQUAL GNU)
  configure_file(gcc.f90.inc gcc.f90 COPYONLY)
  target_sources(pathlib PRIVATE ${CMAKE_CURRENT_BINARY_DIR}/gcc.f90)
elseif(CMAKE_Fortran_COMPILER_ID MATCHES "^Intel")
  configure_file(intel.f90.inc intel.f90 COPYONLY)
  target_sources(pathlib PRIVATE ${CMAKE_CURRENT_BINARY_DIR}/intel.f90)
else()
  configure_file(unknown.f90.inc unknown.f90 COPYONLY)
  target_sources(pathlib PRIVATE ${CMAKE_CURRENT_BINARY_DIR}/unknown.f90)
endif()
